# =============================================================================
# تابع کمکی برای نمایش خوانای رابطه برتری (dominance relation)
# منبع: بخش ۵.۲ کتاب Hyperparameter Tuning with R
# هدف: تبدیل کلاس‌های رابطه به رشته مرتب‌شده مثل "{A_1} > {A_3} > {A_2}"
# =============================================================================

show_relation <- function(r) {
  # استخراج کلاس‌های رابطه (گروه‌های معادل الگوریتم‌ها)
  classes <- relation_classes(r)
  
  # ساخت نام کلاس‌ها به صورت {A_1, A_2} یا {A_3}
  class_names <- sapply(classes, function(x) {
    paste0("{", paste(x, collapse = ", "), "}")
  })
  
  # ترکیب کلاس‌ها با علامت > برای نمایش ترتیب
  paste(class_names, collapse = " > ")
}

# اجرای تابع روی رابطه r_mean (که قبلاً ساخته شده)
show_relation(r_mean)
